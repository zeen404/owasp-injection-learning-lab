SQL Injection

ช่องโหว่ยอดนิยมที่สุด — ผู้โจมตีแทรก SQL code เข้าใน query เพื่อ bypass login, ดึงข้อมูลลับ หรือทำลายฐานข้อมูล

Payloads: ' OR '1'='1 ' UNION SELECT 1,secret,3 FROM secrets-- '; DROP TABLE users;--
Classic UNION Blind Error-based
🔴 Vulnerable 🟢 Secure

Command Injection

แอปส่ง user input ไปยัง OS shell โดยตรง ทำให้ผู้โจมตีรัน shell command ได้อย่างอิสระบนเซิร์ฟเวอร์

Payloads: 127.0.0.1; cat /etc/passwd | whoami 127.0.0.1 && id
Chaining Piping Blind
🔴 Vulnerable 🟢 Secure

🛡 การแก้ไข (Secure)

# Python - Parameterized Query
cursor.execute("SELECT * FROM users WHERE username=%s AND password=%s",
               (username, password))

# PHP - PDO Prepared Statement
$stmt = $pdo->prepare("SELECT * FROM users WHERE username=? AND password=?");
$stmt->execute([$username, $password]);

🛡 การแก้ไข (Secure)

# PHP - escape ก่อนส่ง shell

# Python - ใช้ subprocess.run() กับ list

    ["ping", "-c", "3", ip],
    capture_output=True, text=True, timeout=10
)

# หรือ whitelist validation
if not re.match(r'^\d{1,3}(\.\d{1,3}){3}$', ip):
    return "Invalid IP address"

🛡 การแก้ไข (Secure)

# Python - escape LDAP special chars
import ldap3

def escape_ldap_filter(value):
    # escape: \ * ( ) \x00
    escape_chars = {'\\':'\\5c','*':'\\2a','(':'\\28',
                    ')':'\\29','\x00':'\\00'}
    for k, v in escape_chars.items():
        value = value.replace(k, v)
    return value

# หรือใช้ ldap3.utils.conv.escape_filter_chars()
from ldap3.utils.conv import escape_filter_chars
safe_user = escape_filter_chars(username)

🛡 การแก้ไข (Secure)

# ❌ Vulnerable

# ✅ Secure - ใช้ template file

# ✅ Secure - escape user input
from markupsafe import escape
safe_name = escape(name)
return render_template_string("Hello {{ name }}!", name=safe_name)
